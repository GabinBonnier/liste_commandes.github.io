Étape 1 : Installation et Configuration de Base

1.1 Premiers pas après installation Debian

# Vérifier l'utilisateur actuel
whoami

# Mise à jour du système (en tant que root)
su -
apt update
apt upgrade -y
apt install curl wget nano vim net-tools htop -y

# Ou avec sudo
sudo apt update

sudo apt install curl wget nano vim net-tools htop -y

1.2 Installation Guest Additions VirtualBox

# Installer les dépendances
sudo apt install build-essential dkms linux-headers-$(uname -r) -y

# Vérifier les headers
ls /usr/src/linux-headers-$(uname -r)

# Monter le CD des Guest Additions
sudo mkdir /mnt/cdrom
sudo mount /dev/cdrom /mnt/cdrom

# Installer les Guest Additions
cd /mnt/cdrom
sudo ./VBoxLinuxAdditions.run

# En cas d'erreur
sudo ./VBoxLinuxAdditions.run --nox11

# Redémarrer
sudo reboot

1.3 Configuration réseau - IP fixe

Méthode 1 : systemd-networkd

# Identifier l'interface réseau
ip link show
ifconfig -a

# Créer le fichier de configuration
sudo nano /etc/systemd/network/20-wired.network
Contenu du fichier /etc/systemd/network/20-wired.network : 
[Match]
Name=enp0s3

[Network]
Address=172.16.1.230/24
Gateway=172.16.1.1
DNS=8.8.8.8
DNS=8.8.4.4
# Activer systemd-networkd
sudo systemctl disable networking
sudo systemctl enable systemd-networkd
sudo systemctl start systemd-networkd
sudo reboot

Méthode 2 : /etc/network/interfaces

# Éditer le fichier interfaces
sudo nano /etc/network/interfaces
Ajouter dans /etc/network/interfaces : 
auto enp0s3
iface enp0s3 inet static
    address 172.16.1.230
    netmask 255.255.255.0
    gateway 172.16.1.1
    dns-nameservers 8.8.8.8 8.8.4.4
# Redémarrer le réseau
sudo systemctl restart networking

4.3 Installation PhpMyAdmin

# Installer PhpMyAdmin
sudo apt install phpmyadmin -y

# Configurer Nginx pour PhpMyAdmin
sudo nano /etc/nginx/sites-available/default
Ajouter la configuration PhpMyAdmin : 
location /phpmyadmin {
    root /usr/share/;
    index index.php index.html index.htm;
    location ~ ^/phpmyadmin/(.+\.php)$ {
        try_files $uri =404;
        root /usr/share/;
        fastcgi_pass unix:/var/run/php/php8.2-fpm.sock;
        fastcgi_index index.php;
        fastcgi_param SCRIPT_FILENAME $document_root$fastcgi_script_name;
        include /etc/nginx/fastcgi_params;
    }
}
# Redémarrer Nginx
sudo systemctl restart nginx

Étape 7 : Configuration HTTPS

7.1 Génération des certificats SSL

# Créer le répertoire SSL
sudo mkdir /etc/nginx/ssl

# Générer certificat auto-signé
sudo openssl req -x509 -nodes -days 365 -newkey rsa:2048 \
    -keyout /etc/nginx/ssl/nginx.key \
    -out /etc/nginx/ssl/nginx.crt
Réponses aux questions OpenSSL :
• Country Name: FR
• State: Hauts-de-France
• City: Calais
• Organization Name: SAE
• Organizational Unit: IT
• Common Name: www.nom1-nom2.net
• Email: developer@localhost

7.2 Configuration SSL dans Nginx

# Sauvegarder la configuration actuelle
sudo cp /etc/nginx/sites-available/default /etc/nginx/sites-available/default.backup

# Éditer la configuration
sudo nano /etc/nginx/sites-available/default
Configuration complète Nginx HTTPS : 
# Redirection HTTP vers HTTPS
server {
    listen 80;
    server_name www.nom1-nom2.net;
    return 301 https://$server_name$request_uri;
}

# Configuration HTTPS
server {
    listen 443 ssl;
    server_name www.nom1-nom2.net;
    
    ssl_certificate /etc/nginx/ssl/nginx.crt;
    ssl_certificate_key /etc/nginx/ssl/nginx.key;
    
    root /var/www/html;
    index index.php index.html index.htm;
    charset utf-8;
    
    location / {
        try_files $uri $uri/ =404;
    }
    
    location ~ \.php$ {
        include snippets/fastcgi-php.conf;
        fastcgi_pass unix:/var/run/php/php8.2-fpm.sock;
    }
    
    location /phpmyadmin {
        root /usr/share/;
        index index.php index.html index.htm;
        location ~ ^/phpmyadmin/(.+\.php)$ {
            try_files $uri =404;
            root /usr/share/;
            fastcgi_pass unix:/var/run/php/php8.2-fpm.sock;
            fastcgi_index index.php;
            fastcgi_param SCRIPT_FILENAME $document_root$fastcgi_script_name;
            include /etc/nginx/fastcgi_params;
        }
    }
}
# Tester la configuration
sudo nginx -t

# Redémarrer Nginx
sudo systemctl restart nginx

Étape 9 : Passage en Production

9.1 Sécurisation PHP

# Éditer php.ini pour la production
sudo nano /etc/php/8.2/fpm/php.ini
Modifier ces valeurs pour la production : 
display_errors = Off
expose_php = Off
log_errors = On
error_log = /var/log/php_errors.log
allow_url_fopen = Off
allow_url_include = Off
# Redémarrer PHP-FPM
sudo systemctl restart php8.2-fpm

9.2 Sécurisation des bases de données

# Configurer MySQL pour accès local uniquement
sudo nano /etc/mysql/mysql.conf.d/mysqld.cnf
Modifier dans mysqld.cnf : 
bind-address = 127.0.0.1
# Configurer PostgreSQL
sudo nano /etc/postgresql/*/main/postgresql.conf
Modifier dans postgresql.conf : 
listen_addresses = 'localhost'
# Redémarrer les services
sudo systemctl restart mysql
sudo systemctl restart postgresql

9.3 Configuration du pare-feu

# Installer UFW
sudo apt install ufw -y

# Configurer les règles
sudo ufw default deny incoming
sudo ufw default allow outgoing
sudo ufw allow ssh
sudo ufw allow 'Nginx Full'

# Activer le pare-feu
sudo ufw enable

# Vérifier le statut
sudo ufw status
